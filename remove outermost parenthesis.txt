class Solution:
    def removeOuterParentheses(self, s: str) -> str:
        cnt=0
        st=""
        for i in s:
            if(i=="("):
                cnt+=1
            if(i=="(" and cnt>1):
                st+="("
            if(i==")" and cnt>1):
                st+=")"
            if(i==")"):
                cnt-=1
        return st
